# 代码生成执行规范测试用例（全场景）

> 目的：用于验证 `project.md` 规则与最新代码生成方案（DDL → spec → 模板渲染 + 前端直出）。
> 约束：所有表遵循数据库设计规范（含标准字段、命名规则、索引、字典 SQL）。

## 1. 使用方式（统一流程）

```bash
# 1) DDL → spec（输出 YAML）
java -jar jeecg-codegen-cli.jar \
  --ddl /path/to/ddl.sql \
  --spec-out /path/to/spec.yaml \
  --output /abs/path/jeecg-boot/jeecg-module-system/jeecg-system-biz \
  --frontend-root /abs/path/jeecg-boot/ant-design-vue-jeecg/src/views \
  --jsp-mode one \
  --bussi-package org.jeecg.modules \
  --entity-package demo

# 2) spec → 模板渲染
java -jar jeecg-codegen-cli.jar \
  --input /path/to/spec.yaml \
  --output /abs/path/jeecg-boot/jeecg-module-system/jeecg-system-biz \
  --frontend-root /abs/path/jeecg-boot/ant-design-vue-jeecg/src/views
```

> 说明：`frontendRoot` 开启后，Vue 文件会自动重定向到前端目录（默认不覆盖已有文件）。

## 2. 场景覆盖总览

| 场景 | DDL 表 | jspMode | 备注 |
| --- | --- | --- | --- |
| 单表（标准 CRUD） | `demo_orders` | `one` | 字典、金额、日期、文件/图片、开关 |
| 单表（字段控件覆盖） | `demo_assets` | `one` | markdown/富文本/时间/省市区/URL |
| 树表 | `demo_categories` | `tree` | `parent_id` 树结构字段 |
| 一对多（主从） | `demo_contracts` + `demo_contract_items` | `many` | 默认一对多 |
| 一对多（JVXE/ERP/InnerTable/Tab） | 同上 | `jvxe/erp/innerTable/tab` | 同一套 DDL 不同风格 |
| 弹窗/搜索下拉/部门/用户 | `demo_customers` | `one` | 需在 spec 手动指定控件 |

## 3. SQL 汇总（DDL + 字典 SQL）

```sql
CREATE TABLE `demo_orders` (
  `id` varchar(32) NOT NULL COMMENT '主键ID',
  `del_flag` varchar(1) DEFAULT '0' COMMENT '删除状态 0=未删除 1=已删除',
  `create_by` varchar(32) DEFAULT NULL COMMENT '创建人',
  `create_time` datetime DEFAULT (now()) not null COMMENT '创建时间',
  `update_by` varchar(32) DEFAULT NULL COMMENT '更新人',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',

  `order_no` varchar(64) NOT NULL COMMENT '订单编号',
  `order_name` varchar(120) NOT NULL COMMENT '订单名称',
  `order_status` varchar(1) DEFAULT NULL COMMENT '订单状态：0=待处理；1=处理中；2=已完成；3=已取消。字典：demo_order_status',
  `order_type` varchar(2) DEFAULT NULL COMMENT '订单类型：01=标准；02=加急；03=外包。字典：demo_order_type',
  `customer_id` varchar(32) DEFAULT NULL COMMENT '客户ID(关联 demo_customers 表)',
  `order_amount` decimal(10,2) DEFAULT '0.00' COMMENT '订单金额',
  `order_date` datetime DEFAULT NULL COMMENT '下单时间',
  `delivery_date` date DEFAULT NULL COMMENT '交付日期',
  `is_urgent` tinyint(1) DEFAULT 0 COMMENT '是否加急 0=否 1=是',
  `contact_phone` varchar(20) DEFAULT NULL COMMENT '联系人电话',
  `contact_email` varchar(100) DEFAULT NULL COMMENT '联系人邮箱',
  `attachment_file` varchar(255) DEFAULT NULL COMMENT '附件文件',
  `preview_image` varchar(255) DEFAULT NULL COMMENT '预览图片',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',
  `sort_order` int DEFAULT 0 COMMENT '排序',

  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_demo_orders_order_no` (`order_no`),
  KEY `idx_demo_orders_customer_id` (`customer_id`),
  KEY `idx_demo_orders_order_status` (`order_status`),
  KEY `idx_demo_orders_order_date` (`order_date`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='订单测试表';

CREATE TABLE `demo_customers` (
  `id` varchar(32) NOT NULL COMMENT '主键ID',
  `del_flag` varchar(1) DEFAULT '0' COMMENT '删除状态 0=未删除 1=已删除',
  `create_by` varchar(32) DEFAULT NULL COMMENT '创建人',
  `create_time` datetime DEFAULT (now()) not null COMMENT '创建时间',
  `update_by` varchar(32) DEFAULT NULL COMMENT '更新人',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',

  `customer_code` varchar(64) NOT NULL COMMENT '客户编码',
  `customer_name` varchar(120) NOT NULL COMMENT '客户名称',
  `customer_level` varchar(1) DEFAULT NULL COMMENT '客户级别：A=重点；B=普通；C=潜在。字典：demo_customer_level',
  `customer_status` varchar(1) DEFAULT NULL COMMENT '客户状态：0=停用；1=启用。字典：demo_customer_status',
  `contact_phone` varchar(20) DEFAULT NULL COMMENT '联系人电话',
  `contact_email` varchar(100) DEFAULT NULL COMMENT '联系人邮箱',
  `contact_address` varchar(255) DEFAULT NULL COMMENT '联系地址',
  `owner_user_id` varchar(32) DEFAULT NULL COMMENT '负责人用户ID(关联 sys_user 表)',
  `dept_id` varchar(32) DEFAULT NULL COMMENT '所属部门ID(关联 sys_depart 表)',
  `pca` varchar(255) DEFAULT NULL COMMENT '省市区',
  `website_url` varchar(200) DEFAULT NULL COMMENT '官网地址',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',
  `sort_order` int DEFAULT 0 COMMENT '排序',

  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_demo_customers_customer_code` (`customer_code`),
  KEY `idx_demo_customers_status` (`customer_status`),
  KEY `idx_demo_customers_owner_user_id` (`owner_user_id`),
  KEY `idx_demo_customers_dept_id` (`dept_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='客户测试表';

CREATE TABLE `demo_categories` (
  `id` varchar(32) NOT NULL COMMENT '主键ID',
  `del_flag` varchar(1) DEFAULT '0' COMMENT '删除状态 0=未删除 1=已删除',
  `create_by` varchar(32) DEFAULT NULL COMMENT '创建人',
  `create_time` datetime DEFAULT (now()) not null COMMENT '创建时间',
  `update_by` varchar(32) DEFAULT NULL COMMENT '更新人',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',

  `parent_id` varchar(32) DEFAULT NULL COMMENT '父级ID',
  `category_code` varchar(64) NOT NULL COMMENT '分类编码',
  `category_name` varchar(120) NOT NULL COMMENT '分类名称',
  `category_type` varchar(2) DEFAULT NULL COMMENT '分类类型：01=业务；02=系统。字典：demo_category_type',
  `category_status` varchar(1) DEFAULT NULL COMMENT '分类状态：0=停用；1=启用。字典：demo_category_status',
  `is_leaf` tinyint(1) DEFAULT 0 COMMENT '是否叶子节点 0=否 1=是',
  `level_num` int DEFAULT 1 COMMENT '层级',
  `path` varchar(500) DEFAULT NULL COMMENT '层级路径',
  `sort_order` int DEFAULT 0 COMMENT '排序',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',

  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_demo_categories_code` (`category_code`),
  KEY `idx_demo_categories_parent_id` (`parent_id`),
  KEY `idx_demo_categories_status` (`category_status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='分类树测试表';

CREATE TABLE `demo_contracts` (
  `id` varchar(32) NOT NULL COMMENT '主键ID',
  `del_flag` varchar(1) DEFAULT '0' COMMENT '删除状态 0=未删除 1=已删除',
  `create_by` varchar(32) DEFAULT NULL COMMENT '创建人',
  `create_time` datetime DEFAULT (now()) not null COMMENT '创建时间',
  `update_by` varchar(32) DEFAULT NULL COMMENT '更新人',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',

  `contract_no` varchar(64) NOT NULL COMMENT '合同编号',
  `contract_name` varchar(120) NOT NULL COMMENT '合同名称',
  `customer_id` varchar(32) DEFAULT NULL COMMENT '客户ID(关联 demo_customers 表)',
  `contract_status` varchar(1) DEFAULT NULL COMMENT '合同状态：0=草稿；1=生效；2=完结。字典：demo_contract_status',
  `sign_date` date DEFAULT NULL COMMENT '签订日期',
  `start_date` date DEFAULT NULL COMMENT '开始日期',
  `end_date` date DEFAULT NULL COMMENT '结束日期',
  `contract_amount` decimal(12,2) DEFAULT '0.00' COMMENT '合同金额',
  `is_active` tinyint(1) DEFAULT 1 COMMENT '是否有效 0=否 1=是',
  `attachment_file` varchar(255) DEFAULT NULL COMMENT '合同附件',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',
  `sort_order` int DEFAULT 0 COMMENT '排序',

  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_demo_contracts_contract_no` (`contract_no`),

  KEY `idx_demo_contracts_status` (`contract_status`),
  KEY `idx_demo_contracts_sign_date` (`sign_date`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='合同主表';

CREATE TABLE `demo_contract_items` (
  `id` varchar(32) NOT NULL COMMENT '主键ID',
  `del_flag` varchar(1) DEFAULT '0' COMMENT '删除状态 0=未删除 1=已删除',
  `create_by` varchar(32) DEFAULT NULL COMMENT '创建人',
  `create_time` datetime DEFAULT (now()) not null COMMENT '创建时间',
  `update_by` varchar(32) DEFAULT NULL COMMENT '更新人',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',

  `contract_id` varchar(32) NOT NULL COMMENT '合同ID(关联 demo_contracts 表)',
  `item_code` varchar(64) NOT NULL COMMENT '明细编号',
  `item_name` varchar(120) NOT NULL COMMENT '明细名称',
  `item_unit` varchar(10) DEFAULT NULL COMMENT '单位：PCS=件；SET=套。字典：demo_item_unit',
  `item_price` decimal(10,2) DEFAULT '0.00' COMMENT '单价',
  `item_qty` int DEFAULT 1 COMMENT '数量',
  `item_amount` decimal(12,2) DEFAULT '0.00' COMMENT '金额',
  `delivery_date` date DEFAULT NULL COMMENT '交付日期',
  `item_image` varchar(255) DEFAULT NULL COMMENT '明细图片',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',

  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_demo_contract_items_contract_item` (`contract_id`, `item_code`),
  KEY `idx_demo_contract_items_contract_id` (`contract_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='合同明细表';

CREATE TABLE `demo_assets` (
  `id` varchar(32) NOT NULL COMMENT '主键ID',
  `del_flag` varchar(1) DEFAULT '0' COMMENT '删除状态 0=未删除 1=已删除',

  `update_time` datetime DEFAULT NULL COMMENT '更新时间',

  `asset_code` varchar(64) NOT NULL COMMENT '资产编码',
  `asset_name` varchar(120) NOT NULL COMMENT '资产名称',
  `asset_type` varchar(2) DEFAULT NULL COMMENT '资产类型：01=设备；02=资料；03=证照。字典：demo_asset_type',
  `asset_source` varchar(2) DEFAULT NULL COMMENT '资产来源：01=采购；02=自研。字典：demo_asset_source',
  `asset_desc` text COMMENT '资产描述',
  `asset_content` longtext COMMENT '资产内容',
  `purchase_time` datetime DEFAULT NULL COMMENT '采购时间',
  `available_time` time DEFAULT NULL COMMENT '可用时间',
  `warranty_date` date DEFAULT NULL COMMENT '质保到期日',
  `is_shared` tinyint(1) DEFAULT 0 COMMENT '是否共享 0=否 1=是',
  `manual_file` varchar(255) DEFAULT NULL COMMENT '说明书文件',
  `photo_image` varchar(255) DEFAULT NULL COMMENT '资产图片',
  `asset_url` varchar(200) DEFAULT NULL COMMENT '资产链接',
  `rating_score` decimal(3,1) DEFAULT '0.0' COMMENT '评分',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',
  `sort_order` int DEFAULT 0 COMMENT '排序',

  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_demo_assets_asset_code` (`asset_code`),
  KEY `idx_demo_assets_type` (`asset_type`),
  KEY `idx_demo_assets_source` (`asset_source`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='资产测试表';
```
